\subsection*{Training Convergence}
\label{app:training}

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{fig_training_convergence}
    \caption{Training convergence per curriculum stage (8 stages with $\geq50$ logged
             episodes shown; short interrupted stages omitted). Each line is one stage,
             coloured light-to-dark with stage index. \textbf{Left:} mean episode reward
             --- early stages (4--6) converge at large negative values; later stages
             (7, 10, 13, 15, 16) converge to $3$--$5\times10^6$.
             \textbf{Right:} TD loss --- all stages show the spike-then-converge profile.
             Raw values in transparent colour; EMA ($w=0.9$) in bold.}
    \label{fig:training_convergence}
\end{figure}
