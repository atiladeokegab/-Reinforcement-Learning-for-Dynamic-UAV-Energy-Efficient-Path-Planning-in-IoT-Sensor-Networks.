\begin{frame}{What prior work does -- and misses}

          Optimal SF allocation. Exponential in $N$; intractable $N{>}20$.

% SLIDE 5 --- Problem formulation + why it's hard  (1:30)
% SPEAKER: "The objective is to maximise total data collected from
% all sensors over the episode — that outer double sum. Subject to
% three hard constraints: link quality must exceed the SF threshold,
% the battery must not run out, and no two sensors may share the
% same spreading factor in the same slot — that's the collision
% constraint. These three constraints map directly onto three
% physical forces the drone feels simultaneously: battery decay,
% quartic path-loss, and a protocol that lags the drone's movement
% by ten seconds. No closed form handles all three at once.
% That is why this needs a learned policy."
% ------------------------------------------------------------
\begin{frame}{Problem formulation}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \small
      \emp{Objective:} maximise total data collected
      \[
        \max_{\pi}\;\sum_{t=0}^{T}\sum_{i=1}^{N} D_{i,t}
      \]
      \vspace{-0.5em}
      \emp{Subject to:}
      \begin{alignat}{2}
        &\mathrm{RSSI}_{i,t}(p_t) \ge \mathrm{RSSI}_{\min}(SF_{i,t})
          &\quad&\forall i \notag\\
        &E_{t+1} = E_t - P(a_t)\Delta t \ge 0
          &\quad&\forall t \notag\\
        &\textstyle\sum_{i\in\mathcal{N}}\mathbf{1}[SF_{i,t}{=}k] \le 1
          &\quad&\forall k\in\{7,\ldots,12\} \notag\\
        &p_{t+1} = p_t + v(a_t)\,\Delta t \notag
      \end{alignat}
      \vspace{-0.3em}
      \footnotesize\qt{$D_{i,t}$: bytes from sensor $i$ at step $t$.
      $\pi$: control policy. $E_t$: battery. $SF_{i,t}$: spreading factor.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \small\emp{Three forces act simultaneously:}\\[0.5em]
      \begin{tabular}{@{}p{2.2cm} p{3.8cm}@{}}
        \emp{Battery} &
          $E_{t+1}=E_t-P(a_t)\Delta t$\\
          & \tiny\qt{$P_\text{hover}>P_\text{flight}$; hovering costs more.}\\[0.4em]
        \emp{Coverage} &
          $P_r\propto d^{-4}$\\
          & \tiny\qt{Two-Ray: halve distance $\Rightarrow$ 16$\times$ power.}\\[0.4em]
        \emp{Protocol lag} &
          $\bar{R}_t=\lambda R_t+(1{-}\lambda)\bar{R}_{t-1}$\\
          & \tiny\qt{EMA-ADR, $\lambda{=}0.1$; SF lags ${\sim}$10\,s.}\\
      \end{tabular}
      \vfill
      \begin{block}{}
        \centering\small
        Non-convex. Non-stationary. No closed form.\\
        \qt{$\Rightarrow$ trained a \emp{Deep Q-Network} to find $\pi$.}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------
% SLIDE 10 --- Root cause: representational ceiling  (1:00)
% SPEAKER: "Why? I traced it back to the network's input
% representation. The flat MLP concatenates all sensor features
% into a single fixed-length vector: sensor 1 at positions 1 to 5,
% sensor 2 at positions 6 to 10, and so on. Sensor order is
% baked into the representation. If the sensors appear in a
% different order, the policy sees a completely different input
% and has no way to recognise they are the same sensors.
% At small grids, the network memorises the fixed layout.
% At large grids with more sensors, it cannot generalise — so
% it retreats to a safe default: the perimeter. This is not a
% training failure. It is a representational ceiling."
% ------------------------------------------------------------
\begin{frame}{The cause: representational ceiling}
  \vspace{0.4em}
  \begin{block}{}
    \large\centering Not reward. \quad Not training. \quad Not physics.
  \end{block}
  \vspace{0.8em}
  The flat MLP consumes \(\;[\mathrm{uav},\; s_1,\; s_2,\; \ldots,\; s_N]\;\)
  as a single concatenated vector.\\[0.7em]

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{The problem}
        \small
        Sensor order is part of the representation.\\
        Swapping sensor 1 and sensor 2 $\Rightarrow$ the policy
        sees a completely different input.\\[0.3em]
        At small scales: memorise the fixed layout.\\
        At large scales: cannot generalise.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{The consequence}
        \small
        Perimeter patrol is the cheapest \emph{stable}
        trajectory the network can express under this
        encoding.\\[0.3em]
        \emp{The architecture cannot treat the sensor
        set as a set.}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
